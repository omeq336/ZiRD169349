SELECT last_name || ' '|| salary AS wynagrodzenie
FROM EMPLOYEES
WHERE department_id IN(20,50)
AND salary BETWEEN 2000 AND 7000
ORDER BY last_name

SELECT hire_date, last_name, &user_column
FROM EMPLOYEES
WHERE manager_id IS NOT NULL
AND EXTRACT(YEAR FROM hire_date) == 2005
ORDER BY &user_column;

SELECT first_name || ' ' || last_name AS person, salary, phone_number
FROM EMPLOYEES
WHERE SUBSTR(last_name, 3, 1) = 'e'
AND first_name LIKE '%' || '&partial_name' || '%'
ORDER BY 1 DESC, 2 ASC;

SELECT first_name, last_name, ROUND(MONTHS_BETWEEN(SYSDATE, hire_date)) as amount_of_months,
CASE 
WHEN ROUND(MONTHS_BETWEEN(SYSDATE, hire_date)) <= 150 THEN salary * 0.10
WHEN ROUND(MONTHS_BETWEEN(SYSDATE, hire_date)) <= 200 THEN salary * 0.20
ELSE salary * 0.30
END AS bonus_amount
FROM EMPLOYEES
ORDER BY amount_of_months;

SELECT department_id, SUM(salary) AS income_sum, ROUND(AVG(salary)) AS average_income
FROM employees
GROUP BY deparment_id
