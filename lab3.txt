SELECT 
    employee_id,
    first_name,
    last_name,
    salary,
    DENSE_RANK() OVER (ORDER BY salary DESC) AS ranking
FROM EMPLOYEES;

SELECT 
    employee_id,
    first_name,
    last_name,
    salary,
    DENSE_RANK() OVER (ORDER BY salary DESC) AS ranking,
    SUM(salary) OVER () AS total_salary
FROM EMPLOYEES;

-- zad 3

CREATE TABLE sales AS
SELECT * FROM HR.sales;

CREATE TABLE products AS
SELECT * FROM HR.products;

-- zad 4

SELECT
    e.last_name,
    p.product_name,
    SUM(s.quantity * s.price) OVER (PARTITION BY e.employee_id) AS total_employee_sales,
    DENSE_RANK() OVER (ORDER BY (s.quantity * s.price) DESC) AS sales_rank
FROM sales s
JOIN products p 
    ON s.product_id = p.product_id
JOIN employees e 
    ON s.employee_id = e.employee_id;

-- zad 5

SELECT
    e.last_name,
    p.product_name,
    s.price,
    COUNT(*) OVER (
        PARTITION BY s.product_id, TRUNC(s.sale_date)
    ) AS transactions_per_day,
    SUM(s.quantity * s.price) OVER (
        PARTITION BY s.product_id, TRUNC(s.sale_date)
    ) AS total_amount_per_day,
    LAG(s.price) OVER (
        PARTITION BY s.product_id 
        ORDER BY s.sale_date
    ) AS previous_price,
    LEAD(s.price) OVER (
        PARTITION BY s.product_id 
        ORDER BY s.sale_date
    ) AS next_price
FROM sales s
JOIN products p 
    ON s.product_id = p.product_id
JOIN employees e 
    ON s.employee_id = e.employee_id;
    
-- zad 6

SELECT
    p.product_name,
    s.price,
    SUM(s.quantity * s.price) OVER (
        PARTITION BY s.product_id, TRUNC(s.sale_date, 'MM')
    ) AS total_month_amount,
    SUM(s.quantity * s.price) OVER (
        PARTITION BY s.product_id, TRUNC(s.sale_date, 'MM')
        ORDER BY s.sale_date, s.sale_id
        ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW
    ) AS growing_month_amount
FROM sales s
JOIN products p
    ON s.product_id = p.product_id;

-- zad 7

SELECT
    p.product_name,
    p.product_category,
    s22.price AS price_2022,
    s23.price AS price_2023,
    (s23.price - s22.price) AS price_difference
FROM sales s22
JOIN sales s23
    ON s22.product_id = s23.product_id
    AND TO_CHAR(s22.sale_date, 'MM-DD') = TO_CHAR(s23.sale_date, 'MM-DD')
    AND EXTRACT(YEAR FROM s22.sale_date) = 2022
    AND EXTRACT(YEAR FROM s23.sale_date) = 2023
JOIN products p
    ON p.product_id = s22.product_id;

-- zad 8

SELECT
    p.product_category,
    p.product_name,
    s.price,
    MIN(s.price) OVER (
        PARTITION BY p.product_category
    ) AS min_price_in_category,
    MAX(s.price) OVER (
        PARTITION BY p.product_category
    ) AS max_price_in_category,
    MAX(s.price) OVER (PARTITION BY p.product_category)
    - MIN(s.price) OVER (PARTITION BY p.product_category) AS price_difference
FROM sales s
JOIN products p
    ON s.product_id = p.product_id;

-- zad 9

SELECT
    p.product_name,
    s.sale_date,
    s.price,
    AVG(s.price) OVER (
        PARTITION BY s.product_id
        ORDER BY s.sale_date
        ROWS BETWEEN 1 PRECEDING AND 1 FOLLOWING
    ) AS moving_avg_price
FROM sales s
JOIN products p
    ON s.product_id = p.product_id;

-- zad 10

SELECT
    p.product_name,
    p.product_category,
    s.price,
    RANK() OVER (
        PARTITION BY p.product_category
        ORDER BY s.price DESC
    ) AS rank_in_category,
    ROW_NUMBER() OVER (
        PARTITION BY p.product_category
        ORDER BY s.price DESC
    ) AS row_number_in_category,
    DENSE_RANK() OVER (
        PARTITION BY p.product_category
        ORDER BY s.price DESC
    ) AS dense_rank_in_category
FROM sales s
JOIN products p
    ON s.product_id = p.product_id;

-- zad 11

SELECT
    e.last_name,
    p.product_name,
    (s.price * s.quantity) AS order_value,
    SUM(s.price * s.quantity) OVER (
        PARTITION BY s.employee_id
        ORDER BY s.sale_date, s.sale_id
        ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW
    ) AS running_employee_sales,
    DENSE_RANK() OVER (
        ORDER BY (s.price * s.quantity) DESC
    ) AS global_sales_rank
FROM sales s
JOIN products p
    ON s.product_id = p.product_id
JOIN employees e
    ON s.employee_id = e.employee_id;
